# 背景

因为一篇天涯帖子，进行了爬虫初入门。

# 实现

# 1. 下载与引用

// 下载
// npm i cheerio iconv-lite

// 引用
const cheerio = require('cheerio')
const iconv = require('iconv-lite') // iconv-lite 是解码工具

# 2. 使用步骤

// 引入服务器，如果需要访问的链接是 http 协议，那么请引用 'http'
const https = require('https')

https.get('https://jandan.net/ooxx', function (sres) {
    // 1. 拿到网页内容
    let chunks = [];
    sres.on('data', function (chunk) {
        chunks.push(chunk);
    });
    sres.on('end', function () {
        // 2. 对获取的数据进行解码，根据页面的编码规格来，比如‘utf-8’是这么来的
        // <meta http-equiv="Content-Type" content="text/html; charset=UTF-8">
        let html = iconv.decode(Buffer.concat(chunks), 'utf-8');
            
        // 3. 加载获取的页面数据，实现了 jQuery 接口
        let $ = cheerio.load(html, {
            decodeEntities: false
        });
            
        // 4. 这里可以快乐地进行 DOM 操作
            
    });
});
